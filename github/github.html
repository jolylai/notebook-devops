# SSH Key

# 生成 key
ssh-keygen -t rsa -C "your.email@example.com" -b 4096

# 将 key 复制到黏贴板

# Git Bash on Windows / Windows PowerShell
cat ~/.ssh/id_rsa.pub | clip

# macOS
pbcopy < ~/.ssh/id_rsa.pub

# Linux
xclip -sel clip < ~/.ssh/id_rsa.pub

# Windows
type %userprofile%\.ssh\id_rsa.pub | clip
